### 1 两数之和
```python 
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        dic={}
        for i in range(len(nums)):
            v=target-nums[i]
            if v in dic:
                return [dic[v],i]
            else:
                dic[nums[i]]=i 
```

### 383 赎金信

```python 
class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        from collections import Counter
        a = Counter(ransomNote)
        b = Counter(magazine)
        for i,j in a.items():
            if j > b[i]:
                return False
        return True
```
