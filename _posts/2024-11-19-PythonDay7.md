### 1 两数之和
```python 
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        dic={}
        for i in range(len(nums)):
            v=target-nums[i]
            if v in dic:
                return [dic[v],i]
            else:
                dic[nums[i]]=i 
```


### 454 四数相加II

```python
class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        dic={}
        count =0
        for i in nums1:
            for j in nums2: 
                if i+j not in dic.keys():
                    dic[i+j] = 1
                else:
                    dic[i+j] += 1
        for k in nums3:
            for l in nums4: 
                target = -(k+l)
                if target in dic.keys():
                    count += dic[target]
        return count

```
### 383 赎金信

```python 
class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        from collections import Counter
        a = Counter(ransomNote)
        b = Counter(magazine)
        for i,j in a.items():
            if j > b[i]:
                return False
        return True
```
